void hall_state()
{
	float seconds;
	get_hall();
	motor_state = (hall_a << 2)|(hall_b << 1)|(hall_c);
	if(motor_state == last_motor_state)
	{
		counter.rpm_reset++;
	}
	if((motor_state != last_motor_state)&&(last_motor_state != 0))
	{
		counter.rpm_reset = 0;
		counter.state_change_counter++;
		status.rpm_flag = 1;
		if(counter.state_change_counter == 6)
		{
			turn_value = TIM2->CNT;
			time_interval = turn_value - last_turn_value;
			seconds = (1.0 / 1000000) * (double)time_interval;
			rpm = 60 / seconds;
			last_turn_value = turn_value;
			counter.state_change_counter = 0;
		}
	}
	if(counter.rpm_reset == 500)
	{
		status.rpm_flag = 0;
		counter.rpm_reset=0;
		rpm = 0;
	}
	last_motor_state = motor_state;
}

uint64_t INA229_readReg(uint8_t regAddr)
{
    uint64_t value;
    int i;

    uint8_t txBuf[1] = {0};
    uint8_t rxBuf[6] = {0}; //max buffer size

    txBuf[0] = (regAddr << 2 ) | 0x01;

    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_4, 0);
    HAL_SPI_TransmitReceive(&hspi1, txBuf, rxBuf, 6, 100);
    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_4, 1);

    value = 0;

    for(i = 1; i < INA229_regSize[regAddr]+1; i++)
    {
        value = (value << 8) | rxBuf[i];
    }

    return value;
}
float INA229_getVBUS_V()
{
    uint64_t value = INA229_readReg(0x05);
    float data;

    //Remove reserved bits
    value = value >> 4;

    //Convert for 2's compliment and signed value (though always positive)
    if(value > 0x7FFFF)
    {
        data = (float)value - 0x100000; //left for redundancy and error checking, should never get used
    }
    else
    {
        data = (float)value;
    }

    //Convert to V
    data = (data * 195.3125) / 1000000;

    return data;
}
float INA229_getCURRENT_A()
{
	float currentlsb = 30.2 / 524288;

    float data = INA229_getCURRENT_signedLSB();

    data = data * currentlsb;

    return data;
}
float INA229_getCURRENT_signedLSB()
{
    uint64_t value = INA229_readReg(0x07);
    float data;

    //Remove reserved bits
    value = value >> 4;

    //Convert for 2's compliment and signed value
    if(value > 0x7FFFF)
    {
        data = (float)value - 0x100000;
    }
    else
    {
        data = (float)value;
    }

    return data;
}
void INA229_writeReg(uint8_t regAddr, uint16_t value)
{
    uint8_t txBuf[3] = {0};

    txBuf[0] = regAddr << 2;
    txBuf[1] = MSB(value);
    txBuf[2] = LSB(value);
    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_4, 0);
    HAL_SPI_Transmit(&hspi1, txBuf, 3, 100);
    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_4, 1);
}
